135 	public void paint (final Graphics g){	  
136 		super.paint(g);       
137 		int x = 0;
138 	    int y = 0;
139 	    for (int i = 0; i < 240; i++) {
140 	        x = (i % 20) * (this.getWidth()/20);
141 	        y = ((i - (x / (this.getWidth()/20))) / 20) * (this.getHeight()/12);
142 	        g.drawImage(this.tabmap.get(map.getOnMap(i)), x, y,this.getWidth()/20,this.getHeight()/12, this);
143 	        g.drawString("Score " + String.valueOf(Score.getScore()),500 , 95);
144 	        g.setColor(Color.yellow);  
145 	    }
